11c. Crea una funci√≥n arraySwap(array) que tome una matriz y devuelva una matriz donde el primer y el √∫ltimo valor se intercambian (o se conmutan).
  - arraySwap([1, 20, 22, 24, 5]) // [5, 20, 22, 24, 1]
  - arraySwap(['hola', 'hola', 'bien']) // ['bien', 'hola', 'hola']
SOLUCION:
      function arraySwap(array) {
        /*
        Necesitamos guardar los valores antes de intercambiarlos. 
        De lo contrario, si intercambiamos directamente, perder√≠amos uno de los valores.
        */
        let dato = array[0]  //guarda el primer valor
        array[0] = array[array.length -1];  //intercambia el valor 1 por el ultimo del array
        array[array.length -1] = dato; //intercambia el ultimo valor por el primero que fue guardado

        console.log(array); //muestra en consola el array actualizado

      }

      arraySwap([1, 20, 22, 24, 5]) // [5, 20, 22, 24, 1]
      arraySwap(['hola', 'hello', 'bien']) // ['bien', 'hello', 'hola']
